CREATE SCHAME SAM;

CREATE TABLE CURSO(
    Sigla VARCHAR(10),
    Nome VARCHAR(30) NOT NULL UNIQUE,
    Horas int NOT NULL (Horas > 0),
    Custo float NOT NULL AND DEFAULT 0.00,
    Area VARCHAR(30) NOT NULL,
    CONSTRAINT PK_Curso PRIMARY KEY(Sigla),
    CONSTRAINT FK_CursoArea FOREIGN KEY (Area) REFERENCES AREA(Sigla)
);

CREATE TABLE AREA(
    Sigla VARCHAR(10),
    Nome VARCHAR(30) NOT NULL,
    SuperArea(30) NOT NULL,
    CONSTRAIN PK_Area PRIMARY KEY(Sigla),
    CONSTRAIN FK_SuperAREA FOREIGN KEY(SuperArea) REFERENCES AREA(Sigla)
);

CREATE TABLE MODULO(
    Sigla VARCHAR(10),
    Nome VARCHAR (30) NOT NULL,
    Curso VARCHAR(30) NOT NULL,
    CONSTRAIN PK_Modulo PRIMARY KEY(Sigla),
    CONSTRAINT FK_ModuloCurso FOREIGN KEY (Curso) REFERENCES CURSO(Sigla) ON DELETE CASCADE,
);

CREATE TABLE TOPICO(
    Modulo VARCHAR(10),
    Sigla VARCHAR(10),
    Nome VARCHAR(30) NOT NULL,
    Horas INT NOT NULL (Horas > 0),
    CONSTRAIN FK_Topico PRIMARY KEY(Modulo, Sigla),
    CONSTRAIN FK_Topico FOREIGN KEY(Modulo),
);

CREATE TABLE MATRICULA (
    Curso VARCHAR(10),
    Aluno CHAR(11),
    Data DATE DEFAULT CURRENT_DATE,
    Pago BOOLEAN DEFAULT FALSE,
    CONSTRAINT PK_Matricula PRIMARY KEY (Curso, Aluno),
    CONSTRAINT FK_MatriculaCurso FOREIGN KEY (Curso) REFERENCES CURSO(Sigla),
    CONSTRAINT FK_MatriculaAluno FOREIGN KEY (Aluno) REFERENCES ALUNO(CPF)
);

CREATE TABLE ALUNO (
    CPF CHAR(11),
    Nome VARCHAR(50) NOT NULL,
    Sobrenome VARCHAR(50) NOT NULL,
    Sexo CHAR(1) CHECK (Sexo IN ('M', 'F', 'O')),
    DataNasc DATE NOT NULL,
    CONSTRAINT PK_Aluno PRIMARY KEY(CPF)
);

CREATE TABLE PROFESSOR (
    Curso VARCHAR(10),
    CPF CHAR(11),
    Nome VARCHAR(100) NOT NULL,
    CONSTRAINT PK_Professor PRIMARY KEY (Curso, CPF),
    CONSTRAINT FK_ProfessorCurso FOREIGN KEY (Curso) REFERENCES CURSO(Sigla)
);
