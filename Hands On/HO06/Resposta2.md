CREATE TABLE CURSO(
    Sigla VARCHAR(10) NOT NULL,                                                 // Se você não restringe PK como não nulo o MYSQL da erro na hora de fazer ALTER
    Nome VARCHAR(30) NOT NULL,                                                  // Unique fica no comando ALTER
    Horas INTEGER DEFAULT 1 CHECK (Horas > 0),
    Custo DECIMAL(10, 2) DEFAULT 0.00,
    Area VARCHAR(30) NOT NULL,
);
    ALTER TABLE CURSO ADD CONSTRAINT PK_Curso PRIMARY KEY(Sigla);
    ALTER TABLE CURSO ADD CONSTRAIN UN_NomeCurso UNIQUE(Nome);
    ALTER TABLE CURSO ADD CONSTRAINT FK_CursoArea FOREIGN KEY (Area) REFERENCES AREA(Sigla) ON DELETE RESTRICT ON UPDATE CASCADE;


CREATE TABLE AREA(
    Sigla VARCHAR(10) NOT NULL,

    SuperArea(30),
);
